3   /**
4    * A simple Base64 implementation. This avoids the following issues:
5    * <ul>
6    * <li>Android has it's own Base64 parser that is not exposed to standard Java
7    * <li>using Commons-Codec that is compatible with Android is not OSGi
8    * compliant.
9    * <li>the javax.xml.DatatypeConverter may lose characters when there is missing
10   * padding.
11   * </ul>
12   *
13   * @author Archimedes Trajano
14   */
15  public final class Base64 {
16      /**
17       * Decoding map.
18       */
19      private static final byte[] DECODE_MAP;
20  
21      /**
22       * Encoding map.
23       */
24      private static final char[] ENCODE_MAP = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789-_".toCharArray();
25  
26      static {
27          DECODE_MAP = new byte[128];
28          for (byte b = 0; b < 64; ++b) {
29              DECODE_MAP[ENCODE_MAP[b]] = b;
30          }
31          DECODE_MAP['+'] = 62;
32          DECODE_MAP['/'] = 63;
33      }
34  
35      /**
36       * Decodes a base64 string.
37       *
38       * @param base64String
39       *            base64 string
40       * @return bytes
41       */
42      public static byte[] decode(final String base64String) {
43          final byte[] buffer = new byte[decodeLength(base64String)];
44          decode(base64String, buffer, 0);
45          return buffer;
46      }
47  
48      /**
49       * Decodes a base64 string to a buffer.
50       *
51       * @param base64String
52       *            base64 string
53       * @param buffer
54       *            buffer to fill
55       * @param offset
56       *            offset
57       * @return amount of bytes decoded.
58       */
59      public static int decode(final String base64String, final byte[] buffer, final int offset) {
60          int p = 0;
61          final byte[] base64Chars = base64String.getBytes();
62          for (int i = 0; i < base64Chars.length; ++i) {
63              if (i % 4 == 0) {
64                  buffer[offset + i - p] = (byte) (DECODE_MAP[base64Chars[i]] << 2);
65              } else if (i % 4 == 1 && offset + i - p - 1 < buffer.length) {
66                  buffer[offset + i - p - 1] |= DECODE_MAP[base64Chars[i]] >> 4;
67                  if (offset + i - p < buffer.length) {
68                      buffer[offset + i - p] = (byte) (DECODE_MAP[base64Chars[i]] << 4);
69                  }
70              } else if (i % 4 == 2 && offset + i - p - 1 < buffer.length) {
71                  buffer[offset + i - p - 1] |= DECODE_MAP[base64Chars[i]] >>> 2;
72                  if (offset + i - p < buffer.length) {
73                      buffer[offset + i - p] = (byte) (DECODE_MAP[base64Chars[i]] << 6);
74                  }
75              } else if (i % 4 == 3 && offset + i - p - 1 < buffer.length) {
76                  buffer[offset + i - p - 1] |= DECODE_MAP[base64Chars[i]];
77                  p++;
78              }
79          }
80          return decodeLength(base64String);
81      }
82  
83      /**
84       * Gets the byte length of the decoded text.
85       *
86       * @param base64String
87       *            Base64 string
88       * @return byte length of the decoded text.
89       */
90      public static int decodeLength(final String base64String) {
91          final int originalLength = base64String.length();
92          if (originalLength == 0) {
93              return 0;
94          } else if (base64String.charAt(originalLength - 2) == '=') {
95              return (originalLength - 2) * 3 / 4;
96          } else if (base64String.charAt(originalLength - 1) == '=') {
97              return (originalLength - 1) * 3 / 4;
98          } else {
99              return originalLength * 3 / 4;
100         }
101     }

104      * Encodes bytes into a Base64 string.
105      *
106      * @param bytes
107      *            bytes to encode
108      * @return Base64 string
109      */
110     public static String encode(final byte[] bytes) {
111         return encode(bytes, 0, bytes.length, true);
112     }

114     /**
115      * Encodes bytes in a buffer into a Base64 string with padding.
116      *
117      * @param bytes
118      *            bytes buffer
119      * @param offset
120      *            offset
121      * @param length
122      *            number of bytes to encode
123      * @return Base64 string
124      */
125     public static String encode(final byte[] bytes, final int offset, final int length) {
126         return encode(bytes, offset, length, true);
127     }
128 
129     /**
130      * Encodes bytes in a buffer into a Base64 string.
131      *
132      * @param bytes
133      *            bytes buffer
134      * @param offset
135      *            offset
136      * @param length
137      *            number of bytes to encode
138      * @param padding
139      *            add padding if needed.
140      * @return Base64 string
141      */
142     public static String encode(final byte[] bytes, final int offset, final int length, final boolean padding) {
143         final StringBuilder buffer = new StringBuilder(length * 3);
144         for (int i = offset; i < offset + length; i += 3) {
145             // p's are the segments for each byte. For every triple there are 6
146             // segments
147             int p0 = bytes[i] & 0xFC;
148             p0 >>= 2;
149 
150             int p1 = bytes[i] & 0x03;
151             p1 <<= 4;
152 
153             int p2;
154             int p3;
155             if (i + 1 < offset + length) {
156                 p2 = bytes[i + 1] & 0xF0;
157                 p2 >>= 4;
158                 p3 = bytes[i + 1] & 0x0F;
159                 p3 <<= 2;
160             } else {
161                 p2 = 0;
162                 p3 = 0;
163             }
164             int p4;
165             int p5;
166             if (i + 2 < offset + length) {
167                 p4 = bytes[i + 2] & 0xC0;
168                 p4 >>= 6;
169                 p5 = bytes[i + 2] & 0x3F;
170             } else {
171                 p4 = 0;
172                 p5 = 0;
173             }
174 
175             if (i + 2 < offset + length) {
176                 buffer.append(ENCODE_MAP[p0]);
177                 buffer.append(ENCODE_MAP[p1 | p2]);
178                 buffer.append(ENCODE_MAP[p3 | p4]);
179                 buffer.append(ENCODE_MAP[p5]);
180             } else if (i + 1 < offset + length) {
181                 buffer.append(ENCODE_MAP[p0]);
182                 buffer.append(ENCODE_MAP[p1 | p2]);
183                 buffer.append(ENCODE_MAP[p3]);
184                 if (padding) {
185                     buffer.append('=');
186                 }
187             } else {
188                 buffer.append(ENCODE_MAP[p0]);
189                 buffer.append(ENCODE_MAP[p1 | p2]);
190                 if (padding) {
191                     buffer.append("==");
192                 }
193             }
194         }
195         return buffer.toString();
196     }
197 
198     /**
199      * Encodes bytes into a Base64 string without padding.
200      *
201      * @param bytes
202      *            bytes to encode
203      * @return Base64 string
204      */
205     public static String encodeWithoutPadding(final byte[] bytes) {
206         return encode(bytes, 0, bytes.length, false);
207     }
