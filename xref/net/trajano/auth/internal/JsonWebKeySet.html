12  import javax.json.JsonObject;
13  import javax.json.JsonValue;

15  /**
16   * JSON Web Key Set. Implements
17   * http://tools.ietf.org/html/draft-ietf-jose-json-web-key-29
18   *
19   * TODO when supporting more methods, consider converting this to a composite
20   * pattern.
21   *
22   * @author Archimedes Trajano
23   *
24   */
25  public class JsonWebKeySet {
26      /**
27       * Maps "kid" to the {@link Key}. Instances of this classes are short lived
28       * and used in one thread context so concurrent use is not expected.
29       */
30      private final Map<String, Key> keyMap = new HashMap<>(); // NOPMD
31  
32      /**
33       * Constructs the JSON Web Key using the JSON object. Unsupported key usage
34       * scenarios are silently dropped.
35       *
36       * @param obj
37       *            JSON object
38       * @throws GeneralSecurityException
39       *             problem with the crypto APIs
40       */
41      public JsonWebKeySet(final JsonObject obj) throws GeneralSecurityException {
42          for (final JsonValue v : obj.getJsonArray("keys")) {
43              final JsonObject keyJson = (JsonObject) v;
44              final String kid;
45              if (keyJson.containsKey("kid")) {
46                  kid = keyJson.getString("kid");
47              } else {
48                  kid = "";
49              }
50              if (keyJson.containsKey("use")
51                      && "sig".equals(keyJson.getString("use"))) {
52                  final PublicKey key = buildPublicKey(keyJson);
53                  if (key != null) {
54                      keyMap.put(kid, key);
55                  }
56              }
57          }
58      }
59  
60      /**
61       * This builds the {@link PublicKey} based on the "kty" value. Unsupported
62       * key types are silently dropped.
63       *
64       * @param keyJson
65       *            JSON object containing the key data
66       * @return public key
67       * @throws GeneralSecurityException
68       *             problem with the crypto APIs
69       */
70      private PublicKey buildPublicKey(final JsonObject keyJson)
71              throws GeneralSecurityException {
72          final String kty = keyJson.getString("kty");
73          if ("RSA".equals(kty)) {
74              return buildRSAPublicKey(keyJson);
75          }
76          return null;
77      }
78  
79      /**
80       * Builds an RSA {@link PublicKey}. It uses "n" as the modulus and "e" as
81       * the public exponent.
82       *
83       * @param keyJson
84       *            JSON object containing the key data
85       * @return public key
86       * @throws GeneralSecurityException
87       *             problem with the crypto APIs
88       */
89      private PublicKey buildRSAPublicKey(final JsonObject keyJson)
90              throws GeneralSecurityException {
91          final BigInteger modulus = new BigInteger(1, Base64.decode(keyJson
92                  .getString("n")));
93          final BigInteger publicExponent = new BigInteger(1,
94                  Base64.decode(keyJson.getString("e")));
95          return KeyFactory.getInstance("RSA").generatePublic(
96                  new RSAPublicKeySpec(modulus, publicExponent));
97      }
98  
99      /**
100      * Gets a key given a Key ID.
101      *
102      * @param kid
103      *            Key ID
104      * @return key
105      */
106     public Key getKey(final String kid) {
107         return keyMap.get(kid);
108     }
109 
110     /**
111      * Gets a key given a Key ID with a specified type.
112      *
113      * @param kid
114      *            Key ID
115      * @param clazz
116      *            type of key.
117      * @param <T>
118      *            type
119      * @return key
120      */
121     @SuppressWarnings("unchecked")
122     public <T extends Key> T getKey(final String kid, final Class<T> clazz) {
123         return (T) keyMap.get(kid);
124     }
125 }
