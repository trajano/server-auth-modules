1   package net.trajano.auth.internal;
2   
3   import javax.xml.bind.annotation.XmlAccessType;
4   import javax.xml.bind.annotation.XmlAccessorType;
5   import javax.xml.bind.annotation.XmlElement;
6   import javax.xml.bind.annotation.XmlRootElement;
7   
8   /**
9    * OAuth token.
10   *
11   * @author Archimedes Trajano
12   */
13  @XmlRootElement
14  @XmlAccessorType(XmlAccessType.FIELD)
15  public class OAuthToken {
16      /**
17       * Access token.
18       */
19      @XmlElement(name = "access_token")
20      private String accessToken;
21  
22      /**
23       * Expires in. This is the time in seconds since 1970-01-01T00:00:00Z.
24       */
25      @XmlElement(name = "expires_in")
26      private int expiresIn;
27  
28      /**
29       * ID Token. Used to contain the user's identification.
30       */
31      @XmlElement(name = "id_token")
32      private String idToken;
33  
34      /**
35       * The refresh token, which can be used to obtain new access tokens using
36       * the same authorization grant as described in Section 6.
37       */
38      @XmlElement(name = "refresh_token")
39      private String refreshToken;
40  
41      /**
42       * Token type.
43       */
44      @XmlElement(name = "token_type")
45      private String tokenType;
46  
47      public String getAccessToken() {
48          return accessToken;
49      }
50  
51      public int getExpiresIn() {
52          return expiresIn;
53      }
54  
55      public String getIdToken() {
56          return idToken;
57      }
58  
59      public String getRefreshToken() {
60          return refreshToken;
61      }
62  
63      public String getTokenType() {
64          return tokenType;
65      }
66  
67      public void setAccessToken(final String accessToken) {
68          this.accessToken = accessToken;
69      }
70  
71      public void setExpiresIn(final int expiresIn) {
72          this.expiresIn = expiresIn;
73      }
74  
75      public void setIdToken(final String idToken) {
76          this.idToken = idToken;
77      }
78  
79      public void setRefreshToken(final String refreshToken) {
80          this.refreshToken = refreshToken;
81      }
82  
83      public void setTokenType(final String tokenType) {
84          this.tokenType = tokenType;
85      }
86  
87      @Override
88      public String toString() {
89          return "OAuthToken [accessToken=" + accessToken + ", expiresIn=" + expiresIn + ", idToken=" + idToken + ", refreshToken=" + refreshToken + ", tokenType=" + tokenType + "]";
90      }
91  }
