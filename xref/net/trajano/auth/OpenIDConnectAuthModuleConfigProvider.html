3   import java.util.Map;
4   
5   import javax.security.auth.callback.CallbackHandler;
6   import javax.security.auth.message.AuthException;
7   import javax.security.auth.message.config.AuthConfigFactory;
8   import javax.security.auth.message.config.AuthConfigProvider;
9   import javax.security.auth.message.config.ClientAuthConfig;
10  import javax.security.auth.message.config.ServerAuthConfig;
11  
12  import net.trajano.auth.internal.OpenIDConnectServerAuthConfig;
13  
14  /**
15   * This is used to provide the server auth module on the application rather than
16   * being globally configured in a container.
17   */
18  public class OpenIDConnectAuthModuleConfigProvider implements AuthConfigProvider {
19  
20      /**
21       * {@link AuthConfigFactory} passed in through the constructor. This is not
22       * being used anywhere at the moment.
23       */
24      @SuppressWarnings("unused")
25      private final AuthConfigFactory authConfigFactory;
26  
27      /**
28       * Options.
29       */
30      private final Map<String, String> options;
31  
32      /**
33       * This is called by
34       * {@link AuthConfigFactory#registerConfigProvider(String, Map, String, String, String)}
35       * when registering the provider.
36       *
37       * @param options
38       *            options to pass to the modules and the name of the module
39       *            classes
40       * @param authConfigFactory
41       *            configuration factory
42       */
43      public OpenIDConnectAuthModuleConfigProvider(final Map<String, String> options, final AuthConfigFactory authConfigFactory) {
44  
45          this.authConfigFactory = authConfigFactory;
46          this.options = options;
47      }
48  
49      @Override
50      public ClientAuthConfig getClientAuthConfig(final String layer,
51              final String appContext,
52              final CallbackHandler handler) throws AuthException {
53  
54          return null;
55      }
56  
57      @Override
58      public ServerAuthConfig getServerAuthConfig(final String layer,
59              final String appContext,
60              final CallbackHandler handler) throws AuthException {
61  
62          return new OpenIDConnectServerAuthConfig(options, layer, appContext, handler);
63      }
64  
65      /**
66       * Does nothing.
67       */
68      @Override
69      public void refresh() {
70  
71      }
72  
73  }
