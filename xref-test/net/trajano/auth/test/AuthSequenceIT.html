27      @Test
28      public void testRedirectToEndpoint() throws Exception {
29  
30          final OpenIDConnectAuthModule module = new OpenIDConnectAuthModule();
31  
32          final MessagePolicy mockRequestPolicy = mock(MessagePolicy.class);
33          when(mockRequestPolicy.isMandatory()).thenReturn(true);
34          final Map<String, String> options = ImmutableMap.<String, String> builder()
35                  .put(OpenIDConnectAuthModule.ISSUER_URI_KEY, "https://accounts.google.com")
36                  .put(REDIRECTION_ENDPOINT_URI_KEY, "/app/oauth2")
37                  .put(OAuthModule.CLIENT_ID_KEY, "clientID")
38                  .put(OAuthModule.CLIENT_SECRET_KEY, "clientSecret")
39                  .build();
40          module.initialize(mockRequestPolicy, null, null, options);
41  
42          final MessageInfo messageInfo = mock(MessageInfo.class);
43  
44          final HttpServletRequest servletRequest = mock(HttpServletRequest.class);
45          when(servletRequest.getMethod()).thenReturn("GET");
46          when(servletRequest.getRequestURL()).thenReturn(new StringBuffer("https://i.trajano.net:8443/util/ejb2"));
47          when(servletRequest.getRequestURI()).thenReturn("/util/ejb2");
48          when(servletRequest.getContextPath()).thenReturn("/util");
49          when(servletRequest.isSecure()).thenReturn(true);
50  
51          when(messageInfo.getRequestMessage()).thenReturn(servletRequest);
52  
53          final HttpServletResponse servletResponse = mock(HttpServletResponse.class);
54          when(messageInfo.getResponseMessage()).thenReturn(servletResponse);
55  
56          final Subject client = new Subject();
57          assertEquals(AuthStatus.SEND_CONTINUE, module.validateRequest(messageInfo, client, null));
58          verify(servletResponse).sendRedirect("https://accounts.google.com/o/oauth2/auth?client_id=clientID&response_type=code&scope=openid&redirect_uri=https://i.trajano.net:8443/app/oauth2&state=L2VqYjI");
59      }
