24      @Test
25      public void testCompressDecompress() throws Exception {
26          final String payload = "foo";
27          final ByteArrayOutputStream baos = new ByteArrayOutputStream();
28          final GZIPOutputStream os = new GZIPOutputStream(baos);
29          os.write(payload.toString().getBytes("UTF-8"));
30          System.out.println(baos.toByteArray().length);
31          os.close();
32          final String encoded = Base64.encodeWithoutPadding(baos.toByteArray());
33  
34          final GZIPInputStream is = new GZIPInputStream(
35                  new ByteArrayInputStream(Base64.decode(encoded)));
36          @SuppressWarnings("resource")
37          final String decoded = new Scanner(is).useDelimiter("\\A").next();
38          is.close();
39          assertEquals(payload, decoded);
40      }
